def ناوی_فانکشن(پارامیتەر١, پارامیتەر٢):
    # کۆدی فانکشن
    ئەنجام = پارامیتەر١ + پارامیتەر٢
    return ئەنجام
    
# بانگکردنی فانکشن
نەتیجه = ناوی_فانکشن(5, 3)
print(نەتیجه)  # دەرئەنجام: 8
